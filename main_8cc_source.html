27 static uint8_t selectMode() {
28  auto &sensed = sensor->getSensed();
29  uint8_t mode = 0;
30 
31  auto xLastWakeTime = xTaskGetTickCount();
32  while (true) {
33  vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(50));
34 
35  // 車輪が一定速度以上か一定速度以下で回されたらモード変更
36  if (std::abs(sensed.velocity) > MODE_THRESHOLD_SPEED) {
37  // モード変更
38  mode = (mode + (std::signbit(sensed.velocity) ? -1 : 1)) & 0x0F;
39  }
40  // インジケータ更新
41  for (auto i = 0; i < driver->indicator->counts(); i++) {
42  driver->indicator->set(i, 0, 0, (mode & (0x01 << i)) == 0 ? 0x00 : 0x0F);
43  }
44  driver->indicator->update();
45 
46  // 右壁センサが一定以上ならモード仮確定
47  if (sensed.wall_right90.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_RIGHT90] &&
48  sensed.wall_right45.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_RIGHT45]) {
49  bool blink = false;
50  xLastWakeTime = xTaskGetTickCount();
51  while (true) {
52  vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(50));
53  // 左壁センサが一定以上ならモード確定
54  if (sensed.wall_left90.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_LEFT90] &&
55  sensed.wall_left45.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_LEFT45]) {
56  return mode;
57  }
58 
59  // インジケータ点滅
60  for (auto i = 0; i < driver->indicator->counts(); i++) {
61  uint8_t color = (mode & (0x01 << i)) == 0 ? 0x00 : 0x0F;
62  driver->indicator->set(i, 0, 0, blink ? color : 0x00);
63  }
64  driver->indicator->update();
65  blink = !blink;
66  }
67  }
68  }
