29  uint8_t mode = 0;
30 
31  auto xLastWakeTime = xTaskGetTickCount();

34 
35  // 車輪が一定速度以上か一定速度以下で回されたらモード変更
36  if (std::abs(sensed.velocity) > MODE_THRESHOLD_SPEED) {
37  // モード変更
38  mode = (mode + (std::signbit(sensed.velocity) ? -1 : 1)) & 0x0F;
39  driver->buzzer->tone(C5, 50);
40  }
41  // インジケータ更新
42  for (auto i = 0; i < driver->indicator->counts(); i++) {
43  driver->indicator->set(i, 0, 0, (mode & (0x01 << i)) == 0 ? 0x00 : 0x0F);
44  }
45  driver->indicator->update();
46 
47  // 右壁センサが一定以上ならモード仮確定
48  if (sensed.wall_right90.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_RIGHT90] &&
49  sensed.wall_right45.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_RIGHT45]) {
50  bool blink = false;
51  xLastWakeTime = xTaskGetTickCount();
52  while (true) {
53  vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(50));

55  if (sensed.wall_left90.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_LEFT90] &&
56  sensed.wall_left45.raw > MODE_THRESHOLD_WALL[PARAMETER_WALL_LEFT45]) {
57  driver->buzzer->tone(C5, 100);

59  }
60

64  driver->indicator->set(i, 0, 0, blink ? color : 0x00);
