32  (2.0f * param.acceleration);

34  (param.max_velocity * param.max_velocity - param.end_velocity * param.end_velocity) / (2.0f * param.acceleration);
35 
36  // 加速

38  target.acceleration = sign * param.acceleration;
39  target.max_velocity = sign * param.max_velocity;
40  return false;
41  }
42  // 等速
